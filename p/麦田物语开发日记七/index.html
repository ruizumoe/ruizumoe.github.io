{
    public List<MapData_SO> mapDataSos;

    private void InitTileDetailsMap(MapData_SO mapDataSo)
    {

            {
                
                int tileX = tileProperty.tileCoordinate.x;

                        break;
                    case GridType.PlaceFurniture:
