    foreach (var item in FindObjectsOfType<Item>())
    {
        SceneItem sceneItem = new SceneItem
        {
            itemId = item.itemID,
            itemPos = new SerializableVector3(item.transform.position)
        };

    if (_sceneItemDict.ContainsKey(SceneManager.GetActiveScene().name))

}
