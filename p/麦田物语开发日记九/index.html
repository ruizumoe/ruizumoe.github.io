private void AddNeighborOfCloseNode(Node closeNode)
{
    Vector2Int curPos = closeNode.GetPos();

    {

            var nextPosY = curPos.y + y;
            var validNeighbourNode = GetValidNeighbourNode(nextPosX, nextPosY);
